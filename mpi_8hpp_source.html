132  template<typename ScalarType, typename Symmetry, typename Name>
133  Tensor<ScalarType, Symmetry, Name> Tensor<ScalarType, Symmetry, Name>::receive(const int source) {
134  auto status = MPI_Status();
135  MPI_Probe(source, mpi_tag, MPI_COMM_WORLD, &status);
136  int length;
137  MPI_Get_count(&status, MPI_BYTE, &length);
138  auto data = std::string(length, '\0'); // 这里不需初始化, 但考虑到load和dump本身效率也不高, 无所谓了
139  MPI_Recv(data.data(), length, MPI_BYTE, source, mpi_tag, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
140  auto result = Tensor<ScalarType, Symmetry, Name>().load(data);
141  return result;
142  }

155  template<typename ScalarType, typename Symmetry, typename Name>
156  Tensor<ScalarType, Symmetry, Name> Tensor<ScalarType, Symmetry, Name>::broadcast(const int root) const {
157  const auto& tensor = *this;
158  if (mpi.size == 1) {
159  return tensor.copy(); // rvalue
160  }
161  if (0 > root || root >= mpi.size) {
162  TAT_error("Invalid root rank when mpi broadcast a tensor");
163  }
164  const auto this_fake_rank = (mpi.size + mpi.rank - root) % mpi.size;
165  Tensor<ScalarType, Symmetry, Name> result;
166  // get from father
167  if (this_fake_rank != 0) {
168  const auto father_fake_rank = (this_fake_rank - 1) / 2;
169  const auto father_real_rank = (father_fake_rank + root) % mpi.size;
170  result = receive(father_real_rank);
171  } else {
172  // 自己就是root的话, 会复制一次张量
173  result = tensor.copy();
174  }
175  // send to son
176  const auto left_son_fake_rank = this_fake_rank * 2 + 1;
177  const auto right_son_fake_rank = this_fake_rank * 2 + 2;
178  if (left_son_fake_rank < mpi.size) {
179  const auto left_son_real_rank = (left_son_fake_rank + root) % mpi.size;
180  result.send(left_son_real_rank);
181  }
182  if (right_son_fake_rank < mpi.size) {
183  const auto right_son_real_rank = (right_son_fake_rank + root) % mpi.size;
184  result.send(right_son_real_rank);
185  }
186  return result;
187  }

189  template<typename ScalarType, typename Symmetry, typename Name>
190  template<typename Func>
191  Tensor<ScalarType, Symmetry, Name> Tensor<ScalarType, Symmetry, Name>::reduce(const int root, Func&& function) const {
192  const auto& tensor = *this;
193  if (mpi.size == 1) {
194  return tensor.copy(); // rvalue
195  }
196  if (0 > root || root >= mpi.size) {
197  TAT_error("Invalid root rank when mpi reduce a tensor");
198  }
199  const auto this_fake_rank = (mpi.size + mpi.rank - root) % mpi.size;
200  Tensor<ScalarType, Symmetry, Name> result;
201  // get from son
202  const auto left_son_fake_rank = this_fake_rank * 2 + 1;
203  const auto right_son_fake_rank = this_fake_rank * 2 + 2;
204  if (left_son_fake_rank < mpi.size) {
205  const auto left_son_real_rank = (left_son_fake_rank + root) % mpi.size;
206  result = function(tensor, receive(left_son_real_rank));
207  }
208  if (right_son_fake_rank < mpi.size) {
209  const auto right_son_real_rank = (right_son_fake_rank + root) % mpi.size;
210  // 如果左儿子不存在, 那么右儿子一定不存在, 所以不必判断result是否有效
211  result = function(result, receive(right_son_real_rank));
212  }
213  // pass to father
214  if (this_fake_rank != 0) {
215  const auto father_fake_rank = (this_fake_rank - 1) / 2;
216  const auto father_real_rank = (father_fake_rank + root) % mpi.size;
217  if (left_son_fake_rank < mpi.size) {
218  result.send(father_real_rank);
219  } else {
220  tensor.send(father_real_rank);
221  }
222  }
223  return result;
224  // 子叶为空tensor, 每个非子叶节点为reduce了所有的后代的结果
225  }
