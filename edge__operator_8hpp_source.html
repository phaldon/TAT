528  offsets.push_back(0);
529  }

533  }
534  }
535  } else {

538  }

541  if (!merge_map.empty()) {
542  for (auto& [symmetries_after_transpose, size] : initialize_block_symmetries_with_check(edge_before_merge)) {
543  // convert sym -> target_sym and offsets
544  // and add to map
545  auto symmetries = std::vector<Symmetry>();
546  auto offsets = std::vector<Size>();
547  symmetries.reserve(rank_after_merge);
548  offsets.reserve(rank_after_merge);
549  bool success = true;

551  // [start, end) be merged
552  auto merge_group_symmetries = std::vector<Symmetry>(); // 长度为单个merge前edge数
553  while (position_before_merge < rank_at_transpose && merge_flag[position_before_merge] == position_after_merge) {
554  merge_group_symmetries.push_back(symmetries_after_transpose[position_before_merge]);

556  }

561  symmetries.push_back(this_symmetry);
562  offsets.push_back(this_offset);
563  } else {
564  success = false;
565  break;
566  }
567  } else {
568  symmetries.push_back(merge_group_symmetries.front());
569  offsets.push_back(0);
570  }
571  }

574  }
575  }
576  } else {
577  for (const auto& [symmetries, block] : result.core->blocks) {
578  data_after_transpose_to_destination[symmetries] = {symmetries, std::vector<Size>(rank_after_merge, 0)};
579  }
580  }
581

583  auto split_flag_mark = std::vector<bool>();
584  auto reversed_before_transpose_flag_mark = std::vector<bool>();
585  auto reversed_after_transpose_flag_mark = std::vector<bool>();
586  auto merge_flag_mark = std::vector<bool>();
587  if constexpr (is_fermi) {
588  split_flag_mark.reserve(rank_before_split);
589  reversed_before_transpose_flag_mark.reserve(rank_at_transpose);
590  reversed_after_transpose_flag_mark.reserve(rank_at_transpose);
591  merge_flag_mark.reserve(rank_after_merge);

595  for (auto i = 0; i < rank_before_split; i++) {

597  }

713  } else {

715  }
716  }

731  destination_block.data() + total_destination_offset,

733  plan_destination_to_source,
