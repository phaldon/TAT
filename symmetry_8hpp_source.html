123  [[nodiscard]] static bool get_transpose_parity(const std::vector<Derived>& symmetries, const std::vector<Rank>& transpose_plan) {
124  auto result = false;
125  for (Rank i = 0; i < symmetries.size(); i++) {
126  for (Rank j = i + 1; j < symmetries.size(); j++) {
127  if (transpose_plan[i] > transpose_plan[j]) {
128  result ^= (static_cast<bool>(symmetries[i].fermi % 2) && static_cast<bool>(symmetries[j].fermi % 2));
129  }
130  }
131  }
132  return result;
133  }
143  [[nodiscard]] static bool get_split_merge_parity(
144  const std::vector<Derived>& symmetries, // before merge length
145  const std::vector<Rank>& split_merge_flag, // before merge length
146  const std::vector<bool>& valid_mark) { // after merge length
147  auto result = false;
148  for (Rank split_merge_group_position = 0, split_merge_begin_position = 0, split_merge_end_position = 0;
149  split_merge_group_position < valid_mark.size();
150  split_merge_group_position++) {
151  // split_merge_group_position point to after merge position
152  // begin_position and end_position point to before merge position
153  while (split_merge_end_position < symmetries.size() && split_merge_flag[split_merge_end_position] == split_merge_group_position) {
154  split_merge_end_position++;
155  }
156  if (valid_mark[split_merge_group_position]) {
157  auto sum_of_parity = 0;
158  auto sum_of_parity_square = 0;
159  for (auto position_in_group = split_merge_begin_position; position_in_group < split_merge_end_position; position_in_group++) {
160  auto this_parity = symmetries[position_in_group].fermi;
161  sum_of_parity += this_parity;
162  sum_of_parity_square += this_parity * this_parity;
163  }
164  result ^= static_cast<bool>(((sum_of_parity * sum_of_parity - sum_of_parity_square) / 2) % 2);
165  }
166  split_merge_begin_position = split_merge_end_position;
167  }
168  return result;
169  }
