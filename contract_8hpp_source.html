287  template<typename ScalarType, typename Symmetry, typename Name>
288  Tensor<ScalarType, Symmetry, Name> contract_without_fuse(
289  const Tensor<ScalarType, Symmetry, Name>& tensor_1,
290  const Tensor<ScalarType, Symmetry, Name>& tensor_2,
291  std::set<std::tuple<Name, Name>> contract_names) {
292  constexpr bool is_fermi = is_fermi_symmetry_v<Symmetry>;
293  constexpr bool is_no_symmetry = std::is_same_v<Symmetry, NoSymmetry>;
294  // 为未来split做准备
295  const Rank rank_1 = tensor_1.names.size();
296  const Rank rank_2 = tensor_2.names.size();
297  // 删除不存在的名称, 即在name tuple list中但不在names中
298  for (auto iterator = contract_names.begin(); iterator != contract_names.end();) {
299  auto found_1 = tensor_1.name_to_index.find(std::get<0>(*iterator));
300  auto found_2 = tensor_2.name_to_index.find(std::get<1>(*iterator));
301  if (found_1 == tensor_1.name_to_index.end() || found_2 == tensor_2.name_to_index.end()) {
302  iterator = contract_names.erase(iterator);
303  } else {
304  ++iterator;
305  }
306  }
307  const auto common_rank = contract_names.size();
308  std::set<Name> contract_names_1;
309  std::set<Name> contract_names_2;
310  for (const auto& [name_1, name_2] : contract_names) {
311  contract_names_1.insert(name_1);
312  contract_names_2.insert(name_2);
313  }
314  if (contract_names_1.size() != common_rank || contract_names_2.size() != common_rank) {
315  TAT_error("Duplicated Contract Name");
316  }
317  // Rank contract_origin_rank = contract_names_1.size();
318  // for (Rank i = 0; i < contract_origin_rank; i++) {
319  // }
320  // 需要反转成 - + - -
321  // 事后恢复两侧的边
322  auto reversed_set_1 = std::set<Name>(); // 第一个张量merge时反转表
323  auto reversed_set_2 = std::set<Name>(); // 第二个张量merge时反转表
324  auto edge_result = std::vector<Edge<Symmetry>>(); // 无对称性的时候不需要split方案直接获取最后的edge
325  if constexpr (is_no_symmetry) {
326  edge_result.reserve(rank_1 + rank_2 - 2 * common_rank);
327  }
328  auto split_map_result = std::map<Name, std::vector<std::tuple<Name, BoseEdge<Symmetry>>>>(); // split方案
329  auto reversed_set_result = std::set<Name>(); // 最后split时的反转标
330  auto name_result = std::vector<Name>(); // 最后split后的name
331  name_result.reserve(rank_1 + rank_2 - 2 * common_rank);
332  split_map_result[InternalName<Name>::Contract_1].reserve(rank_1 - common_rank);
333  split_map_result[InternalName<Name>::Contract_2].reserve(rank_2 - common_rank);
334  auto free_name_1 = std::vector<Name>(); // 第一个张量的自由边, merge时使用
335  free_name_1.reserve(rank_1 - common_rank);
336  for (Rank i = 0; i < rank_1; i++) {
337  const auto& n = tensor_1.names[i];
338  if (find_in_contract_names<0>(contract_names, n) == contract_names.end()) {
339  // 是free name
340  free_name_1.push_back(n);
341  if constexpr (is_no_symmetry) {
342  edge_result.push_back(tensor_1.core->edges[i]);
343  } else {
344  split_map_result.at(InternalName<Name>::Contract_1).push_back({n, {tensor_1.core->edges[i].map}});
345  }
346  name_result.push_back(n);
347  if constexpr (is_fermi) {
348  if (tensor_1.core->edges[i].arrow) {
349  reversed_set_1.insert(n);
350  reversed_set_result.insert(n);
351  }
352  }
353  } else {
354  // 将被contract掉
355  if constexpr (is_fermi) {
356  if (!tensor_1.core->edges[i].arrow) {
357  reversed_set_1.insert(n);
358  }
359  }
360  }
361  }
362  const auto free_rank_1 = free_name_1.size();
363  auto free_name_2 = std::vector<Name>(); // 第二个张量的自由边, merge时使用
364  free_name_2.reserve(rank_2 - common_rank);
365  for (Rank i = 0; i < rank_2; i++) {
366  const auto& n = tensor_2.names[i];
367  if (find_in_contract_names<1>(contract_names, n) == contract_names.end()) {
368  // 是free name
369  free_name_2.push_back(n);
370  if constexpr (is_no_symmetry) {
371  edge_result.push_back(tensor_2.core->edges[i]);
372  } else {
373  split_map_result.at(InternalName<Name>::Contract_2).push_back({n, {tensor_2.core->edges[i].map}});
374  }
375  name_result.push_back(n);
376  if constexpr (is_fermi) {
377  if (tensor_2.core->edges[i].arrow) {
378  reversed_set_2.insert(n);
379  reversed_set_result.insert(n);
380  }
381  }
382  } else {
383  // 将被contract掉
384  if constexpr (is_fermi) {
385  if (tensor_2.core->edges[i].arrow) {
386  reversed_set_2.insert(n);
387  }
388  }
389  }
390  }
391  const auto free_rank_2 = free_name_2.size();
392  // 确定转置方案
393  auto common_name_1 = std::vector<Name>(); // 第一个张量的公共边, merge时使用
394  auto common_name_2 = std::vector<Name>(); // 第二个张量的公共边, merge时使用
395  common_name_1.reserve(common_rank);
396  common_name_2.reserve(common_rank);
397  bool put_common_1_right;
398  bool put_common_2_right;
399  auto fit_tensor_1_common_edge = [&]() {
400  for (const auto& n : tensor_1.names) {
401  if (auto position = find_in_contract_names<0>(contract_names, n); position != contract_names.end()) {
402  common_name_1.push_back(std::get<0>(*position));
403  common_name_2.push_back(std::get<1>(*position));
404  }
405  }
406  };
407  auto fit_tensor_2_common_edge = [&]() {
408  for (const auto& n : tensor_2.names) {
409  if (auto position = find_in_contract_names<1>(contract_names, n); position != contract_names.end()) {
410  common_name_1.push_back(std::get<0>(*position));
411  common_name_2.push_back(std::get<1>(*position));
412  }
413  }
414  };
415  // 确定方案
416  if (free_rank_1 == 0) {
417  put_common_1_right = true;
418  fit_tensor_2_common_edge();
419  put_common_2_right = common_name_2.empty() || common_name_2.back() == tensor_2.names.back();
420  } else if (free_rank_2 == 0) {
421  put_common_2_right = true;
422  fit_tensor_1_common_edge();
423  put_common_1_right = common_name_1.empty() || common_name_1.back() == tensor_1.names.back();
424  } else if (free_name_1.back() != tensor_1.names.back()) {
425  put_common_1_right = true;
426  fit_tensor_1_common_edge();
427  put_common_2_right = common_name_2.empty() || common_name_2.back() == tensor_2.names.back();
428  } else if (free_name_2.back() != tensor_2.names.back()) {
429  put_common_2_right = true;
430  fit_tensor_2_common_edge();
431  put_common_1_right = common_name_1.empty() || common_name_1.back() == tensor_1.names.back();
432  } else {
433  put_common_1_right = false;
434  put_common_2_right = false;
435  fit_tensor_2_common_edge();
436  // 所以尽量大张量放在后侧
437  }
438  // 确定交错的对称性
439  auto delete_1 = std::map<Name, std::map<Symmetry, Size>>();
440  auto delete_2 = std::map<Name, std::map<Symmetry, Size>>();
441  if constexpr (!is_no_symmetry) {
442  for (Rank i = 0; i < common_rank; i++) {
443  auto name_1 = common_name_1[i];
444  auto name_2 = common_name_2[i];
445  auto edge_1 = tensor_1.core->edges[tensor_1.name_to_index.at(name_1)];
446  auto edge_2 = tensor_2.core->edges[tensor_2.name_to_index.at(name_2)];
447  auto delete_unused_dimension = [](const auto& edge_this, const auto& edge_other, const auto& name_this, auto& delete_this) {
448 #ifndef _MSVC_LANG
449  // 2020.10.27 现在版本的msvc认为constexpr需要捕获, 等待新版本msvc支持此功能, 才可以删掉这个判断
450  if constexpr (is_fermi) {
451  if (edge_this.arrow == edge_other.arrow) {
452  TAT_error("Different Fermi Arrow to Contract");
453  }
454  }
455 #endif
456  auto delete_map = std::map<Symmetry, Size>();
457  for (const auto& [symmetry, dimension] : edge_this.map) {
458  auto found = edge_other.map.find(-symmetry);
459  if (found != edge_other.map.end()) {
460  // found
461  if (const auto dimension_other = found->second; dimension_other != dimension) {
462  TAT_error("Different Dimension to Contract");
463  }
464  } else {
465  // not found
466  delete_map[symmetry] = 0;
467  // 用于merge时cut, cut成0会自动删除
468  }
469  }
470  if (!delete_map.empty()) {
471  delete_this[name_this] = std::move(delete_map);
472  }
473  };
474  delete_unused_dimension(edge_1, edge_2, name_1, delete_1);
475  delete_unused_dimension(edge_2, edge_1, name_2, delete_2);
476  }
477  }
478  // merge
479  // 仅对第一个张量的公共边的reverse和merge做符号
480  auto tensor_1_merged = tensor_1.edge_operator(
481  {},
482  {},
483  reversed_set_1,
484  {{InternalName<Name>::Contract_1, free_name_1}, {InternalName<Name>::Contract_2, common_name_1}},
485  put_common_1_right ? std::vector<Name>{InternalName<Name>::Contract_1, InternalName<Name>::Contract_2} :
486  std::vector<Name>{InternalName<Name>::Contract_2, InternalName<Name>::Contract_1},
487  false,
488  {{{}, std::set<Name>(common_name_1.begin(), common_name_1.end()), {}, {InternalName<Name>::Contract_2}}},
489  delete_1);
490  auto tensor_2_merged = tensor_2.edge_operator(
491  {},
492  {},
493  reversed_set_2,
494  {{InternalName<Name>::Contract_2, free_name_2}, {InternalName<Name>::Contract_1, common_name_2}},
495  put_common_2_right ? std::vector<Name>{InternalName<Name>::Contract_2, InternalName<Name>::Contract_1} :
496  std::vector<Name>{InternalName<Name>::Contract_1, InternalName<Name>::Contract_2},
497  false,
498  {{{}, {}, {}, {}}},
499  delete_2);
500  // calculate_product
501  auto product_result = Tensor<ScalarType, Symmetry, Name>(
502  {InternalName<Name>::Contract_1, InternalName<Name>::Contract_2},
503  {std::move(tensor_1_merged.core->edges[!put_common_1_right]), std::move(tensor_2_merged.core->edges[!put_common_2_right])});
504  // 因取了T1和T2的edge，所以会自动去掉merge后仍然存在的交错边
505  auto common_edge = std::move(tensor_1_merged.core->edges[put_common_1_right]);
506 
507  auto max_batch_size = product_result.core->blocks.size();
508  vector<char> transpose_a_list(max_batch_size), transpose_b_list(max_batch_size);
509  vector<int> m_list(max_batch_size), n_list(max_batch_size), k_list(max_batch_size), lda_list(max_batch_size), ldb_list(max_batch_size),
510  ldc_list(max_batch_size);
511  vector<ScalarType> alpha_list(max_batch_size), beta_list(max_batch_size);
512  vector<const ScalarType*> a_list(max_batch_size), b_list(max_batch_size);
513  vector<ScalarType*> c_list(max_batch_size);
514  int batch_size = 0;
515 
516  for (auto& [symmetries, data] : product_result.core->blocks) {
517  // m k n
518  auto symmetries_1 = put_common_1_right ? symmetries : std::vector<Symmetry>{symmetries[1], symmetries[0]};
519  auto symmetries_2 = put_common_2_right ? std::vector<Symmetry>{symmetries[1], symmetries[0]} : symmetries;
520  const auto& data_1 = tensor_1_merged.core->blocks.at(symmetries_1);
521  const auto& data_2 = tensor_2_merged.core->blocks.at(symmetries_2);
522  const int m = product_result.core->edges[0].map.at(symmetries[0]);
523  const int n = product_result.core->edges[1].map.at(symmetries[1]);
524  const int k = common_edge.map.at(symmetries[1]);
525  ScalarType alpha = 1;
526  if constexpr (is_fermi) {
527  // 因为并非标准- + - -产生的符号
528  if ((put_common_2_right ^ !put_common_1_right) && bool(symmetries[0].fermi % 2)) {
529  alpha = -1;
530  }
531  }
532  const ScalarType beta = 0;
533  if (m && n && k) {
534  transpose_a_list[batch_size] = put_common_2_right ? 'T' : 'N';
535  transpose_b_list[batch_size] = put_common_1_right ? 'N' : 'T';
536  m_list[batch_size] = n;
537  n_list[batch_size] = m;
538  k_list[batch_size] = k;
539  alpha_list[batch_size] = alpha;
540  a_list[batch_size] = data_2.data();
541  lda_list[batch_size] = put_common_2_right ? k : n;
542  b_list[batch_size] = data_1.data();
543  ldb_list[batch_size] = put_common_1_right ? k : m;
544  beta_list[batch_size] = beta;
545  c_list[batch_size] = data.data();
546  ldc_list[batch_size] = n;
547  batch_size++;
548  } else if (m && n) {
549  std::fill(data.begin(), data.end(), 0);
550  }
551  }
552  gemm_batch<ScalarType>(
553  transpose_a_list.data(),
554  transpose_b_list.data(),
555  m_list.data(),

557  k_list.data(),
558  alpha_list.data(),

630  if (!(edge_1 == edge_2)) {

656  // 是free name或fuse name
657  if (fuse_names.find(n) == fuse_names.end()) {

662  }
663  }

687  common_name_2.push_back(std::get<1>(*position));
688  }

696  } else if (free_rank_2 == 0) {
