66  template<typename ScalarType>
67  std::istream& scan_complex(std::istream& in, std::complex<ScalarType>& value) {
68  ScalarType part;
69  in >> part;
70  char maybe_i = in.peek();
71  if (maybe_i == 'i') {
72  in.get();
73  // no real part
74  value = std::complex<ScalarType>{0, part};
75  } else {
76  // have real part
77  if (maybe_i == '+' || maybe_i == '-') {
78  // have imag part
79  ScalarType another_part;
80  in >> another_part;
81  value = std::complex<ScalarType>{part, another_part};
82  if (in.get() != 'i') {
83  in.setstate(std::ios::failbit);
84  }
85  } else {
86  // no imag part
87  value = std::complex<ScalarType>{part, 0};
88  }
89  }
90  return in;
91  }
