30  Tensor<ScalarType, Symmetry, Name>::multiple(const SingularType& S, const Name& name, char direction, bool division) const {
31  auto guard = multiple_guard();
32  bool different_direction;
33  if (direction == 'u' || direction == 'U') {
34  different_direction = false;
35  } else if (direction == 'v' || direction == 'V') {
36  different_direction = true;
37  } else {
38  return copy();
39  }
40  const auto found = name_to_index.find(name);
41  if (found == name_to_index.end()) {
42  TAT_warning_or_error_when_multiple_name_missing("Edge not Found in Multiple");
43  return copy();
44  }
45  auto result = same_shape();
46  auto index = found->second;
47  for (const auto& [symmetries, block_source] : core->blocks) {
48  auto& block_destination = result.core->blocks.at(symmetries);
49  auto symmetry_of_s = symmetries[index];
50  if (different_direction) {
51  symmetry_of_s = -symmetry_of_s;
52  }
53 #ifdef TAT_USE_SINGULAR_MATRIX
54  const auto& vector_in_S = S.core->blocks.at({-symmetry_of_s, symmetry_of_s});
55  auto dimension = S.core->edges[1].map.at(symmetry_of_s);
56  auto dimension_plus_one = dimension + 1;
57 #else
58  const auto& vector_in_S = S.value.at(symmetry_of_s);
59  auto dimension = vector_in_S.size();
60 #endif
61  Rank i = 0;
62  Size m = 1;
63  for (; i < index; i++) {
64  m *= core->edges[i].map.at(symmetries[i]);
65  }
66  Size k = core->edges[i].map.at(symmetries[i]);
67  Size n = 1;
68  for (i++; i < names.size(); i++) {
69  n *= core->edges[i].map.at(symmetries[i]);
70  }
71  if (dimension != k) {
72  TAT_error("Vector Size incompatible in Multiple with a tensor");
73  }
74  const auto* data_source = block_source.data();
75  auto* data_destination = block_destination.data();
76 
77  if (division) {
78  for (Size a = 0; a < m; a++) {
79  for (Size b = 0; b < k; b++) {
80 #ifdef TAT_USE_SINGULAR_MATRIX
81  auto v = vector_in_S[b * dimension_plus_one];
82 #else
83  auto v = vector_in_S[b];
84 #endif
85  for (Size c = 0; c < n; c++) {
86  *(data_destination++) = *(data_source++) / v;
87  }
88  }
89  }
90  } else {
91  for (Size a = 0; a < m; a++) {
92  for (Size b = 0; b < k; b++) {
93 #ifdef TAT_USE_SINGULAR_MATRIX
94  auto v = vector_in_S[b * dimension_plus_one];
95 #else
96  auto v = vector_in_S[b];
97 #endif
98  for (Size c = 0; c < n; c++) {
99  *(data_destination++) = *(data_source++) * v;
100  }
101  }
102  }
103  }
104  }
105  return result;
106  }

109  template<typename ScalarType, typename Symmetry, typename Name>
110  Tensor<ScalarType, Symmetry, Name> Tensor<ScalarType, Symmetry, Name>::conjugate() const {
111  auto guard = conjugate_guard();
112  if constexpr (std::is_same_v<Symmetry, NoSymmetry> && is_real_v<ScalarType>) {
113  return copy();
114  }
115  auto result_edges = std::vector<Edge<Symmetry>>();
116  result_edges.reserve(names.size());
117  for (const auto& edge : core->edges) {
118  auto& result_edge = result_edges.emplace_back();
119  if constexpr (is_fermi_symmetry_v<Symmetry>) {
120  result_edge.arrow = !edge.arrow;
121  }
122  for (const auto& [symmetry, dimension] : edge.map) {
123  result_edge.map[-symmetry] = dimension;
124  }
125  }
126  auto transpose_flag = std::vector<Rank>(names.size(), 0);
127  auto valid_flag = std::vector<bool>(1, true);
128  auto result = Tensor<ScalarType, Symmetry, Name>(names, result_edges);
129  for (const auto& [symmetries, block] : core->blocks) {
130  auto result_symmetries = std::vector<Symmetry>();
131  for (const auto& symmetry : symmetries) {
132  result_symmetries.push_back(-symmetry);
133  }
134  // result.core->blocks.at(result_symmetries) <- block
135  const Size total_size = block.size();
136  ScalarType* destination = result.core->blocks.at(result_symmetries).data();
137  const ScalarType* source = block.data();
138  bool parity = false;
139  if constexpr (is_fermi_symmetry_v<Symmetry>) {
140  parity = Symmetry::get_split_merge_parity(symmetries, transpose_flag, valid_flag);
141  }
142  if constexpr (is_complex_v<ScalarType>) {
143  if (parity) {
144  for (Size i = 0; i < total_size; i++) {
145  destination[i] = -std::conj(source[i]);
146  }
147  } else {
148  for (Size i = 0; i < total_size; i++) {
149  destination[i] = std::conj(source[i]);
150  }
151  }
152  } else {
153  if (parity) {
154  for (Size i = 0; i < total_size; i++) {
155  destination[i] = -source[i];
156  }
157  } else {
158  for (Size i = 0; i < total_size; i++) {
159  destination[i] = source[i];
160  }
161  }
162  }
163  }
164  return result;
165  }

169  void set_to_identity(ScalarType* pointer, const std::vector<Size>& dimension, const std::vector<Size>& leading, Rank rank) {
170  auto current_index = std::vector<Size>(rank, 0);
171  while (true) {
172  *pointer = 1;
173  Rank active_position = rank - 1;
174 
175  current_index[active_position]++;
176  pointer += leading[active_position];
177 
178  while (current_index[active_position] == dimension[active_position]) {
179  current_index[active_position] = 0;
180  pointer -= dimension[active_position] * leading[active_position];
181 
182  if (active_position == 0) {
183  return;
184  }
185  active_position--;
186 
187  current_index[active_position]++;
188  pointer += leading[active_position];
189  }
190  }
191  }

193  template<typename ScalarType, typename Symmetry, typename Name>
194  Tensor<ScalarType, Symmetry, Name> Tensor<ScalarType, Symmetry, Name>::identity(const std::set<std::tuple<Name, Name>>& pairs) const {
195  auto rank = names.size();
196  auto half_rank = rank / 2;
197  auto ordered_pair = std::vector<std::tuple<Name, Name>>();
198  auto ordered_pair_index = std::vector<std::tuple<Rank, Rank>>();
199  ordered_pair.reserve(half_rank);
200  ordered_pair_index.reserve(half_rank);
201  auto valid_index = std::vector<bool>(rank, true);
202  for (Rank i = 0; i < rank; i++) {
203  if (valid_index[i]) {
204  const auto& name_to_find = names[i];
205  const Name* name_correspond = nullptr;
206  for (const auto& [name_1, name_2] : pairs) {
207  if (name_1 == name_to_find) {
208  name_correspond = &name_2;
209  break;
210  }
211  if (name_2 == name_to_find) {
212  name_correspond = &name_1;
213  break;
214  }
215  }
216  ordered_pair.push_back({name_to_find, *name_correspond});
217  auto index_correspond = name_to_index.at(*name_correspond);
218  ordered_pair_index.push_back({i, index_correspond});
219  valid_index[index_correspond] = false;
220  }
221  }
222 
223  auto result = same_shape().zero();
224 
225  for (auto& [symmetries, block] : result.core->blocks) {
226  auto dimension = std::vector<Size>(rank);
227  auto leading = std::vector<Size>(rank);
228  for (Rank i = rank; i-- > 0;) {
229  dimension[i] = core->edges[i].map[symmetries[i]];
230  if (i == rank - 1) {
231  leading[i] = 1;
232  } else {
233  leading[i] = leading[i + 1] * dimension[i + 1];
234  }
235  }
236  auto pair_dimension = std::vector<Size>();
237  auto pair_leading = std::vector<Size>();
238  pair_dimension.reserve(half_rank);
239  pair_leading.reserve(half_rank);
240  for (Rank i = 0; i < half_rank; i++) {
241  pair_dimension.push_back(dimension[std::get<0>(ordered_pair_index[i])]);
242  pair_leading.push_back(leading[std::get<0>(ordered_pair_index[i])] + leading[std::get<1>(ordered_pair_index[i])]);
243  // ordered_pair_index使用较大的leading进行从大到小排序，所以pair_leading一定降序
244  }
245  set_to_identity(block.data(), pair_dimension, pair_leading, half_rank);
246  }
247 
248  return result;
249  }

251  template<typename ScalarType, typename Symmetry, typename Name>
252  Tensor<ScalarType, Symmetry, Name> Tensor<ScalarType, Symmetry, Name>::exponential(const std::set<std::tuple<Name, Name>>& pairs, int step) const {
253  real_base_t<ScalarType> norm_max = norm<-1>();
254  auto temporary_tensor_rank = 0;
255  real_base_t<ScalarType> temporary_tensor_parameter = 1;
256  while (temporary_tensor_parameter * norm_max > 1) {
257  temporary_tensor_rank += 1;
258  temporary_tensor_parameter *= 1. / 2;
259  }
260  auto temporary_tensor = *this * temporary_tensor_parameter;
261 
262  auto result = identity(pairs);
263 
264  auto power_of_temporary_tensor = Tensor<ScalarType, Symmetry, Name>();
265 
266  ScalarType series_parameter = 1;
267  for (auto i = 1; i <= step; i++) {
268  series_parameter /= i;
269  if (i == 1) {
270  result += temporary_tensor;
271  } else if (i == 2) {
272  power_of_temporary_tensor = temporary_tensor.contract(temporary_tensor, pairs);
273  // result += series_parameter * power_of_temporary_tensor;
274  result = series_parameter * power_of_temporary_tensor + result;
275  // power_of_temporary_tensor相乘一次后边应该就会稳定, 这个时候将result放在+的右侧, 会使得result边的排列和左侧一样
276  // 从而在 i>2 的时候减少转置
277  } else {
278  power_of_temporary_tensor = power_of_temporary_tensor.contract(temporary_tensor, pairs);
279  result += series_parameter * power_of_temporary_tensor;
280  }
281  }
282 
283  for (auto i = 0; i < temporary_tensor_rank; i++) {
284  result = result.contract(result, pairs);
285  }
286  return result;
287  }
