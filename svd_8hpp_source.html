97  template<typename ScalarType, typename Symmetry, typename Name>
98  [[nodiscard]] Tensor<ScalarType, Symmetry, Name> singular_to_tensor(const std::map<Symmetry, vector<real_base_t<ScalarType>>>& singular) {
99  auto symmetries = std::vector<Edge<Symmetry>>(2);
100  for (const auto& [symmetry, values] : singular) {
101  auto dimension = values.size();
102  symmetries[0].map[-symmetry] = dimension;
103  symmetries[1].map[symmetry] = dimension;
104  }
105  if constexpr (is_fermi_symmetry_v<Symmetry>) {
106  symmetries[0].arrow = false;
107  symmetries[1].arrow = true;
108  }
109  auto result = Tensor<ScalarType, Symmetry, Name>({InternalName<Name>::SVD_U, InternalName<Name>::SVD_V}, std::move(symmetries));
110  for (auto& [symmetries, data_destination] : result.core->blocks) {
111  const auto& data_source = singular.at(symmetries[1]);
112  auto dimension = data_source.size();
113  auto dimension_plus_one = dimension + 1;
114  std::fill(data_destination.begin(), data_destination.end(), 0);
115  for (Size i = 0; i < data_source.size(); i++) {
116  data_destination[i * dimension_plus_one] = data_source[i];
117  }
118  }
119  return result;
120  }

133  template<typename ScalarType>
134  void calculate_svd(
135  const int& m,
136  const int& n,
137  const int& min,
138  const int& max,
139  const ScalarType* a,
140  ScalarType* u,
141  real_base_t<ScalarType>* s,
142  ScalarType* vt) {
143  auto kernel_guard = svd_kernel_guard();
144  // 经过初步测试m > n比m < n和false要好, true是显然更糟糕的
145  if (m > n) {
146  auto new_a = vector<ScalarType>(n * m);
147  auto old_u = vector<ScalarType>(n * min);
148  auto old_vt = vector<ScalarType>(min * m);
149  // new_a = a^T
150  // u s vt = a
151  // vt^T s u^T = a^T
152  // old_u = vt^T
153  // old_vt = u^T
154  matrix_transpose(m, n, a, new_a.data()); // m*n -> n*m
155  calculate_svd_kernel(n, m, min, max, new_a.data(), old_u.data(), s, old_vt.data());
156  matrix_transpose(n, min, old_u.data(), vt); // n*min -> min*n
157  matrix_transpose(min, m, old_vt.data(), u); // min*m -> m*min
158  } else {
159  calculate_svd_kernel(m, n, min, max, a, u, s, vt);
160  }
161  }

249  template<typename ScalarType, typename Symmetry, typename Name>
250  typename Tensor<ScalarType, Symmetry, Name>::svd_result
251  Tensor<ScalarType, Symmetry, Name>::svd(const std::set<Name>& free_name_set_u, const Name& common_name_u, const Name& common_name_v, Size cut)
252  const {
253  auto guard = svd_guard();
254  // free_name_set_u不需要做特殊处理即可自动处理不准确的边名
255  constexpr bool is_fermi = is_fermi_symmetry_v<Symmetry>;
256  const auto rank = names.size();
257  // merge
258  auto free_name_u = std::vector<Name>();
259  auto free_name_v = std::vector<Name>();
260  auto reversed_set_u = std::set<Name>();
261  auto reversed_set_v = std::set<Name>();
262  auto reversed_set_origin = std::set<Name>();
263  auto result_name_u = std::vector<Name>();
264  auto result_name_v = std::vector<Name>();
265  auto free_names_and_edges_u = std::vector<std::tuple<Name, BoseEdge<Symmetry, true>>>();
266  auto free_names_and_edges_v = std::vector<std::tuple<Name, BoseEdge<Symmetry, true>>>();
267  free_name_u.reserve(rank);
268  free_name_v.reserve(rank);
269  result_name_u.reserve(rank + 1);
270  result_name_v.reserve(rank + 1);
271  free_names_and_edges_u.reserve(rank);
272  free_names_and_edges_v.reserve(rank);
273  result_name_v.push_back(common_name_v);
274  for (Rank i = 0; i < names.size(); i++) {
275  const auto& n = names[i];
276  if (free_name_set_u.find(n) != free_name_set_u.end()) {
277  free_name_u.push_back(n);
278  result_name_u.push_back(n);
279  free_names_and_edges_u.push_back({n, {core->edges[i].map}});
280  if constexpr (is_fermi) {
281  if (core->edges[i].arrow) {
282  reversed_set_u.insert(n);
283  reversed_set_origin.insert(n);
284  }
285  }
286  } else {
287  free_name_v.push_back(n);
288  result_name_v.push_back(n);
289  free_names_and_edges_v.push_back({n, {core->edges[i].map}});
290  if constexpr (is_fermi) {
291  if (core->edges[i].arrow) {
292  reversed_set_v.insert(n);
293  reversed_set_origin.insert(n);
294  }
295  }
296  }
297  }
298  result_name_u.push_back(common_name_u);
299  const bool put_v_right = free_name_v.empty() || free_name_v.back() == names.back();
300  auto tensor_merged = edge_operator(
301  {},
302  {},
303  reversed_set_origin,
304  {{InternalName<Name>::SVD_U, free_name_u}, {InternalName<Name>::SVD_V, free_name_v}},
305  put_v_right ? std::vector<Name>{InternalName<Name>::SVD_U, InternalName<Name>::SVD_V} :
306  std::vector<Name>{InternalName<Name>::SVD_V, InternalName<Name>::SVD_U});
307  // tensor -> SVD_U -O- SVD_V
308  // call GESVD
309  auto common_edge_1 = Edge<Symmetry>();
310  auto common_edge_2 = Edge<Symmetry>();
311  // arrow always false
312  for (const auto& [sym, _] : tensor_merged.core->blocks) {
313  auto m = tensor_merged.core->edges[0].map.at(sym[0]);
314  auto n = tensor_merged.core->edges[1].map.at(sym[1]);
315  auto k = m > n ? n : m;
316  common_edge_1.map[sym[1]] = k;
317  common_edge_2.map[sym[0]] = k;
318  }
319  auto tensor_1 = Tensor<ScalarType, Symmetry, Name>{
320  put_v_right ? std::vector<Name>{InternalName<Name>::SVD_U, InternalName<Name>::SVD_V} :
321  std::vector<Name>{InternalName<Name>::SVD_V, InternalName<Name>::SVD_U},
322  {std::move(tensor_merged.core->edges[0]), std::move(common_edge_1)}};
323  auto tensor_2 = Tensor<ScalarType, Symmetry, Name>{
324  put_v_right ? std::vector<Name>{InternalName<Name>::SVD_U, InternalName<Name>::SVD_V} :
325  std::vector<Name>{InternalName<Name>::SVD_V, InternalName<Name>::SVD_U},
326  {std::move(common_edge_2), std::move(tensor_merged.core->edges[1])}};
327  auto result_s = std::map<Symmetry, vector<real_base_t<ScalarType>>>();
328  for (const auto& [symmetries, block] : tensor_merged.core->blocks) {
329  auto* data_u = tensor_1.core->blocks.at(symmetries).data();
330  auto* data_v = tensor_2.core->blocks.at(symmetries).data();
331  const auto* data = block.data();
332  const int m = tensor_1.core->edges[0].map.at(symmetries[0]);
333  const int n = tensor_2.core->edges[1].map.at(symmetries[1]);
334  const int k = m > n ? n : m;
335  const int max = m > n ? m : n;
336  auto s = vector<real_base_t<ScalarType>>(k);
337  auto* s_data = s.data();
338  if (m * n != 0) {
339  calculate_svd<ScalarType>(m, n, k, max, data, data_u, s_data, data_v);
340  }
341  result_s[symmetries[put_v_right]] = std::move(s);
342  }
343 
344  // 分析cut方案
345  Size total_dimension = 0;
346  for (const auto& [symmetry, vector_s] : result_s) {
347  total_dimension += vector_s.size();
348  }
349  auto remain_dimension_u = std::map<Symmetry, Size>();
350  auto remain_dimension_v = std::map<Symmetry, Size>();
351  if (cut != Size(-1) && cut < total_dimension) {
352  // auto remain_dimension = std::map<Symmetry, Size>();
353  for (const auto& [symmetry, vector_s] : result_s) {
354  remain_dimension_u[symmetry] = 0;
355  remain_dimension_v[-symmetry] = 0;
356  }
357  for (Size i = 0; i < cut; i++) {
358  Symmetry maximum_position;
359  real_base_t<ScalarType> maximum_singular = 0;
360  for (const auto& [symmetry, vector_s] : result_s) {
361  if (auto& this_remain = remain_dimension_u.at(symmetry); this_remain != vector_s.size()) {
362  if (auto this_singular = vector_s[this_remain]; this_singular > maximum_singular) {
363  maximum_singular = this_singular;
364  maximum_position = symmetry;
365  }
366  }
367  }
368  remain_dimension_u.at(maximum_position) += 1;
369  remain_dimension_v.at(-maximum_position) += 1;
370  }
371 
372  for (const auto& [symmetry, this_remain] : remain_dimension_u) {
373  if (this_remain == 0) {
374  result_s.erase(symmetry);
375  } else {
376  result_s.at(symmetry).resize(this_remain);
377  }
378  }
379  }
380 
381  const auto& tensor_u = put_v_right ? tensor_1 : tensor_2;
382  const auto& tensor_v = put_v_right ? tensor_2 : tensor_1;
383  // 始终在tensor_1处无符号reverse, 然后判断是否再在tensor_u和tensor_v中分别有无符号翻转
384  // tensor_1 == tensor_u -> u nr // put_v_right
385  // tensor_1 == tensor_v -> v nr v nr u yr -> u yr
386  if constexpr (is_fermi) {
387  reversed_set_u.insert(common_name_u);
388  }
389  // 这里会自动cut
390  auto u = tensor_u.template edge_operator<true>(
391  {{InternalName<Name>::SVD_V, common_name_u}},
392  {{InternalName<Name>::SVD_U, free_names_and_edges_u}},
393  reversed_set_u,
394  {},
395  result_name_u,
396  false,
397  {{{}, put_v_right ? std::set<Name>{} : std::set<Name>{common_name_u}, {}, {}}},
398  {{InternalName<Name>::SVD_V, remain_dimension_u}});
399  auto v = tensor_v.template edge_operator<true>(
400  {{InternalName<Name>::SVD_U, common_name_v}},
401  {{InternalName<Name>::SVD_V, free_names_and_edges_v}},
402  reversed_set_v,
403  {},
404  result_name_v,
405  false,
406  {{{}, {}, {}, {}}},
407  {{InternalName<Name>::SVD_U, remain_dimension_v}});
408  return {
409  std::move(u),
410 #ifdef TAT_USE_SINGULAR_MATRIX
411  singular_to_tensor<ScalarType, Symmetry, Name>(result_s),
412 #else
413  {std::move(result_s)},
414 #endif
415  std::move(v)};
416  }
