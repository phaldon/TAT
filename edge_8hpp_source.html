274  template<typename T, typename F1, typename F2, typename F3>
275  void loop_edge(const T* edges, const Rank rank, F1&& rank0, F2&& dims0, F3&& operate) {
276  if (rank == 0) {
277  rank0();
278  return;
279  }
280  using Symmetry = typename T::symmetry_type;
281  using MapIteratorList = std::vector<typename T::edge_map::const_iterator>;
282  auto symmetry_iterator_list = MapIteratorList();
283  symmetry_iterator_list.reserve(rank);
284  for (auto i = 0; i != rank; ++i) {
285  const auto& map = edges[i].map;
286  auto symmetry_iterator = map.begin();
287  if (symmetry_iterator == map.end()) {
288  dims0();
289  return;
290  }
291  symmetry_iterator_list.push_back(symmetry_iterator);
292  }
293  Rank minimum_changed = 0;
294  while (true) {
295  minimum_changed = operate(symmetry_iterator_list, minimum_changed);
296  auto edge_position = rank - 1;
297  ++symmetry_iterator_list[edge_position];
298  while (symmetry_iterator_list[edge_position] == edges[edge_position].map.end()) {
299  if (edge_position == 0) {
300  return;
301  }
302  symmetry_iterator_list[edge_position] = edges[edge_position].map.begin();
303  --edge_position;
304  ++symmetry_iterator_list[edge_position];
305  }
306  minimum_changed = minimum_changed < edge_position ? minimum_changed : edge_position;
307  }
308  }
