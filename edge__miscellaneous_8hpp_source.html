57  template<typename ScalarType, typename Symmetry, typename Name>
58  Tensor<ScalarType, Symmetry, Name> Tensor<ScalarType, Symmetry, Name>::merge_edge(
59  std::map<Name, std::vector<Name>> merge,
60  const bool apply_parity,
61  const std::set<Name>& parity_exclude_name_merge,
62  const std::set<Name>& parity_exclude_name_reverse) const {
63  // delete edge from names_before_merge if not exist
64  for (auto& [name_after_merge, names_before_merge] : merge) {
65  auto new_names_before_merge = std::vector<Name>();
66  new_names_before_merge.reserve(names_before_merge.size());
67  for (const auto& i : names_before_merge) {
68  if (auto found = name_to_index.find(i); found != name_to_index.end()) {
69  new_names_before_merge.push_back(i);
70  }
71  }
72  names_before_merge.swap(new_names_before_merge);
73  // if len(names_before_merge) == 0, it create a trivial edge?
74  // 不, 在反转target_name中, 如果原merge列表为空则不产生新的target_name
75  // 根据edge_operator中的操作实际上就是消除了这个merge
76  }
77  std::vector<Name> target_name;
78  target_name.reserve(names.size());
79  for (auto iterator = names.rbegin(); iterator != names.rend(); ++iterator) {
80  // 找到且最后 -> 添加新的
81  // 找到不最后 -> 不做事
82  // 没找到 -> 添加
83  auto found_in_merge = false;
84  for (const auto& [name_after_merge, names_before_merge] : merge) {
85  if (auto position_in_group = std::find(names_before_merge.begin(), names_before_merge.end(), *iterator);
86  position_in_group != names_before_merge.end()) {
87  if (*iterator == names_before_merge.back()) {
88  target_name.push_back(name_after_merge);
89  }
90  found_in_merge = true;
91  break;
92  }
93  }
94  if (!found_in_merge) {
95  target_name.push_back(*iterator);
96  }
97  }
98  // 翻转target_name
99  for (const auto& [name_after_merge, names_before_merge] : merge) {
100  if (names_before_merge.empty()) {
101  target_name.push_back(name_after_merge);
102  }
103  }
104  std::reverse(target_name.begin(), target_name.end());
105  return edge_operator(
106  {}, {}, {}, merge, std::move(target_name), apply_parity, {{{}, {}, parity_exclude_name_reverse, parity_exclude_name_merge}});
107  }
